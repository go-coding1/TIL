## 아직 다 못품 예시는 맞는데 틀렸다고 나옴 힝 슬퍼

```java
import java.io.IOException;
import java.util.Scanner;

//대회 개최 12915번
/*
* 대회 열여면 3개(쉬움, 중간, 어려움)으로 구성되어야 돼회 오픈 가능
* 난이도는 5개
* 열수있는 최대 대회를 찾아야함
* 0번 부터 문제를 다 합했을때 가능*/
public class Main {
    public static void main(String [] args) throws IOException {
        Scanner scan = new Scanner(System.in);
        //E,EM,M,MH,H 입력 가능
        int E = scan.nextInt();
        int EM = scan.nextInt();
        int M = scan.nextInt();
        int MH = scan.nextInt();
        int H = scan.nextInt();

        //최소 대회 개최 수 : 0
        //최대 대회 개최 수 : 다 합한것보다 많을수 없음 -> e+em+m+mh+h
        int low = 0;
        int high = E+EM+M+MH+H;
        int mid = 0;
        while(low <= high){
            mid = (low+high)/2;
            if(checkContestAble(mid,E,EM,M,MH,H)){
                high = mid-1;
            }else{
                low = mid+1;
            }
        }
        System.out.println((low+high)/2);
    }

    private static boolean checkContestAble(int mid, int e, int em, int m, int mh, int h) {
        boolean flag = true;
        for(int i=0;i<mid;i++){
            //초급문제 할당
            if(e <= 0){
                em--;
            }else{
                e--;
            }
            //고급문제 할당
            if(h <= 0){
                mh--;
            }else{
                h--;
            }
            //중간 문제 할당
            if(m<=0){
                if(em<=0){
                    mh--;
                }else{
                    em--;
                }
            }else {
                m--;
            }
        }
        if(e<0 ||em<0||m<0||mh<0||h <0)
            return true;
        else
            return false;
    }
}
```
