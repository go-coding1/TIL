이거 아직 제대로 못 품

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.*;

public class Main {
    private static int [] switches;
    public static void main(String[] args) throws IOException {
        Scanner scanner = new Scanner(System.in);

        int switchCount = scanner.nextInt();
        switches = new int[switchCount];
        for(int i=0;i<switchCount;i++){
            switches[i] = scanner.nextInt();
        }
        int people = scanner.nextInt();
        for(int i=0;i<people;i++){
            int sex = scanner.nextInt();
            int num = scanner.nextInt();
            if(sex == 1)
                changeMan(num);
            else
                changeWoman(num);
        }
        for(int i=0;i<switchCount;i++){
            System.out.print(switches[i]+" ");
            if(i%20 == 0 && i !=0)
                System.out.println();
        }
    }
    private static void changeWoman(int num) {
        changeLight(num);
        int left = num - 1;
        int right = num + 1;
        while(true){
            if(left<1 || right > switches.length)
                break;
            if(switches[left-1]==switches[right-1]){
                changeLight(left);
                changeLight(right);
            }
            --left;
            ++right;
        }
    }
    private static void changeMan(int num) {
        int size = switches.length;
        int index = 1;
        while(true){
            if(num*index > size)
                break;
            changeLight(num*index);
            index++;
        }
    }
    private static void changeLight(int num){
        num = num-1;
        if(switches[num] == 1)
            switches[num] = 0;
        else
            switches[num] = 1;
    }
}
```
